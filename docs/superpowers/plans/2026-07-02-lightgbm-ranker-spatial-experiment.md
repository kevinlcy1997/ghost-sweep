def group_precision_at_k(
    predictions,
    k: int,
    score_col: str = "score",
    label_col: str = "actual",
    group_col: str = "target_time",
) -> float:
    """Return mean per-group precision@k across all groups."""
    if k <= 0:
        raise ValueError("k must be positive")
    if predictions.empty:
        return 0.0

    values: list[float] = []
    for _, group in predictions.groupby(group_col, sort=False):
        top = group.sort_values(score_col, ascending=False).head(k)
        if top.empty:
            values.append(0.0)
            continue
        values.append(float(top[label_col].astype(int).mean()))
    return float(sum(values) / len(values)) if values else 0.0


def group_recall_at_k(
    predictions,
    k: int,
    score_col: str = "score",
    label_col: str = "actual",
    group_col: str = "target_time",
) -> float:
    """Return mean per-positive-group recall@k; groups without positives are excluded."""
    if k <= 0:
        raise ValueError("k must be positive")
    if predictions.empty:
        return 0.0

    values: list[float] = []
    for _, group in predictions.groupby(group_col, sort=False):
        positives = int(group[label_col].astype(int).sum())
        if positives == 0:
            continue
        top_hits = int(group.sort_values(score_col, ascending=False).head(k)[label_col].astype(int).sum())
        values.append(float(top_hits / positives))
    return float(sum(values) / len(values)) if values else 0.0
```

def _prepare_ranker_training_frame(
    train_sample: pd.DataFrame,
    target_col: str,
) -> tuple[pd.DataFrame, list[int]]:
    """Sort ranker data by target_time and keep only query groups with positives."""
    if train_sample.empty:
        return train_sample.copy(), []

    frame = train_sample.copy()
    frame["target_time"] = pd.to_datetime(frame["target_time"])
    positive_groups = (
        frame.groupby("target_time")[target_col]
        .sum()
        .loc[lambda values: values > 0]
        .index
    )
    frame = frame[frame["target_time"].isin(set(positive_groups))]
    frame = frame.sort_values(["target_time", target_col, "zone_id"], ascending=[True, False, True])
    group_sizes = frame.groupby("target_time", sort=False).size().astype(int).tolist()
    return frame, group_sizes
```

def _fit_spatial_candidate(
    candidate: Candidate,
    train_sample: pd.DataFrame,
    feature_cols: list[str],
    target_col: str,
) -> Pipeline | None:
    pipeline = _make_spatial_pipeline(candidate)
    if candidate.kind == "ranker":
        ranker_sample, group_sizes = _prepare_ranker_training_frame(train_sample, target_col)
        if ranker_sample.empty or not group_sizes:
            return None
        pipeline.fit(
            ranker_sample[feature_cols],
            ranker_sample[target_col],
            model__group=group_sizes,
        )
        return pipeline

    pipeline.fit(train_sample[feature_cols], train_sample[target_col])
    return pipeline


def _predict_spatial_candidate_scores(
    pipeline: Pipeline,
    candidate: Candidate,
    frame: pd.DataFrame,
    feature_cols: list[str],
) -> np.ndarray:
    if candidate.kind == "ranker":
        scores = pipeline.predict(frame[feature_cols])
    else:
        scores = pipeline.predict_proba(frame[feature_cols])[:, 1]
    return np.asarray(scores, dtype=float)
```

def _operational_spatial_metrics(
    frame: pd.DataFrame,
    target_col: str,
    scores: np.ndarray,
) -> dict[str, float]:
    if frame.empty or "target_time" not in frame:
        return {
            "group_precision_at_50": 0.0,
            "group_recall_at_50": 0.0,

    return {
        "group_precision_at_50": group_precision_at_k(
            scored,
            50,
            score_col="score",
            label_col="actual",

        ),

import pandas as pd

rows = []
summary = pd.read_csv("analysis/two_stage_summary_latest.csv")
for slug in ["30m", "1h", "2h"]:
    folds = pd.read_csv(f"analysis/spatial_model_folds_{slug}_latest.csv")
    medians = folds.groupby("model").median(numeric_only=True).reset_index()
    for _, row in medians.iterrows():
        rows.append(
            {
                "horizon": slug,
                "model": row["model"],
                "cv_neighbor_hit_rate_at_50": row.get("neighbor_hit_rate_at_50", 0.0),
                "cv_group_precision_at_50": row.get("group_precision_at_50", 0.0),
                "cv_group_recall_at_50": row.get("group_recall_at_50", 0.0),
                "cv_precision_at_50": row.get("precision_at_50", 0.0),
                "cv_average_precision": row.get("average_precision", 0.0),
                "cv_top_decile_lift": row.get("top_decile_lift", 0.0),
            }
        )

out = pd.DataFrame(rows)
holdout_cols = [
    "horizon",
    "spatial_model",
    "spatial_neighbor_hit_rate_at_50",
    "spatial_group_precision_at_50",
    "spatial_group_recall_at_50",
    "spatial_precision_at_50",
    "spatial_average_precision",
    "spatial_top_decile_lift",
]
print("Holdout selected models:")
print(summary[[column for column in holdout_cols if column in summary.columns]].to_string(index=False))
out.to_csv("analysis/spatial_ranker_experiment_comparison_latest.csv", index=False)
print(out.to_string(index=False))
PY
```

Expected: `analysis/spatial_ranker_experiment_comparison_latest.csv` exists and includes all candidate models.
